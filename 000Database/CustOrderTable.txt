CREATE TABLE CustOrder (OrderId INT UNSIGNED NOT NULL AUTO_INCREMENT,
CustId INT UNSIGNED NOT NULL,
OrderDate DATE NOT NULL,
TotalCost DECIMAL(7,2) UNSIGNED NOT NULL,
CardNum DECIMAL(16,0) UNSIGNED NOT NULL,
CardSec DECIMAL(4,0) UNSIGNED NOT NULL,
BillAdd VARCHAR(100) NOT NULL,
ShipAdd VARCHAR(100) NOT NULL,
PRIMARY KEY (OrderId),
FOREIGN KEY (CustId) REFERENCES Customer(CustId));
